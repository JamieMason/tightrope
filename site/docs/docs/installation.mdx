## Usage

There are no
[barrel](https://steven-lemon182.medium.com/are-typescript-barrel-files-an-anti-pattern-72a713004250)
[files](https://github.com/vercel/next.js/issues/12557) in tightrope, each
function is imported individually:

```ts
import { pipe } from 'tightrope/fn/pipe';
import { filter } from 'tightrope/gen/filter';
import { map } from 'tightrope/gen/map';
import { range } from 'tightrope/gen/range';
import { reduceEach } from 'tightrope/gen/reduce-each';
import { take } from 'tightrope/gen/take';
import { isEvenNumber } from 'tightrope/guard/is-even-number';
import { multiply } from 'tightrope/number/multiply';

pipe(
  range(0, Infinity),
  map(multiply(2)),
  filter(isEvenNumber),
  take(4),
  reduceEach((sum, value) => sum + value, 0),
  Array.from,
);
// [ 0, 2, 6, 12 ]
```

```ts
pipe(
  range(0, Infinity),

  filter(isEvenNumber),
  take(4),

// [ 0, 2, 6, 12 ]
```
